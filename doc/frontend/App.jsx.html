/**
 * Class App component.
 * @class
 */
class App extends React.Component {
	constructor(props) {
		super(props);
		this.state = {
			outputState: 0 /* 0 = idle, 1 = calculating, 2 = valid response, 3 = error */,
			error: '',
			classification: '',
			summary: '',
		};
		this.updateOutputState = this.updateOutputState.bind(this);
		this.setCalculating = this.setCalculating.bind(this);
	}

	/**
	* Function that takes Flask response and deterines how to set next state to: valid or error
	* @function updateOutputState
	* @param {object} response
	*/
	updateOutputState(response) {
		if (response.error == 'None') {
			this.setState({
				outputState: 2 /* valid response */,
				error: '',
				classification: response.classification,
				summary: response.summary,
			});
		} else {
			this.setState({
				outputState: 3 /* error */,
				error: response.error,
				classification: '',
				summary: '',
			});
		}
	}
	/**
	* Function that sets state to default idle state
	* @function setCalculating
	*/
	setCalculating() {
		this.setState({
			outputState: 0 /* calculating */,
			error: '',
			classification: '',
			summary: '',
		});
	}
	/**
	* Function that sets state to default idle state
	* @function render
	* @return {JSX.Element} - returns a screen that contains both the input and output components
	* the user interacts with the input component to affect the output component
	*/
	render() {
		return (
			<Container>
				{globalStyles}
				<HeaderBox>
					<HeaderTop>
						<Logo src="logo.png" />
					</HeaderTop>
					<HeaderBottom>
						<Headline>How ethical is your end user license agreement?</Headline>
					</HeaderBottom>
				</HeaderBox>
				<BodyBox>
					<BodyBoxHalf>
						<Input updateOutput={this.updateOutputState} setCalculating={this.setCalculating} />
					</BodyBoxHalf>
					<BodyBoxHalf>
						<Output updateResponse={this.state} />
					</BodyBoxHalf>
				</BodyBox>
			</Container>
		);
	}
}
